\begin{frame}
\frametitle{斐波那契数列（Fibonacci sequence）}
\framesubtitle{该数列是由意大利数学家莱昂纳多·斐波那契（Leonardo Fibonacci）在1202年提出的。 \textit{别称：兔子数列。}}

\begin{definition}
\[
\begin{aligned}
&F(0)=0 \\
&F(1)=1 \\
&F(2)=2 \\
&\cdots \\
&F(n)=F(n-1)+F(n-2)
\end{aligned}
\]
\end{definition}

\begin{examples}
\[
1, 2, 3, 5, 8, \cdots
\]
\end{examples}
